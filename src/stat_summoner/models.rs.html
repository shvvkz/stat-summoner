/// Modal pour entrer le nom du joueur et le tag
#[derive(Debug, Modal)]
#[name = "Enter your League of Legends Stats Info"]

    #[name = "Game Name"]
    #[placeholder = "Enter your game name (e.g., Faker)"]
    pub game_name: String,

    #[name = "Tag Line"]
    #[placeholder = "Enter your tag line (e.g., 1234)"]
    pub tag_line: String,
}
